#### 验证微信服务器

> 测试号 > 接口配置信息

```js
const config = {
    token,appID,appsecret
}
app.use((req,res,next)=>{
    const {
        signature,
        echostr,
        timestamp,
        nonce
    } = req.query
    const arr = [timestamp,nonce,token].sort()
    const str = sha1(arr.join(''))
    if(signature === str){
        res.send(echostr)
    }else{
        res.send('error')
    }
})
```
