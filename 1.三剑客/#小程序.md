> 服务器连接微信后台

```js
// 微信密钥|token配置
module.exports={
  appid:'',
  appsecret:'',
  token:''
}
// 服务器拦截
const wechat = require('co-wechat')
const conf = require('./conf')
router.all('/wechat',wechat(conf).middleware(
  async message=>{
    return message.Content
  }
))
```
