### 案例

~~~shell
# 邮箱
reg = /^[\w-]+@([\w-]+\.)+(com|cn|cc|org|net)$/i
# 手机号
reg = /^((0\d{2,3}-\d{7,8}|1[3-9]\d{9}))$/
# 用户名
reg = /^[a-z][\w-]{2,7}$/i
# 密码
reg = /^[a-z0-9]{5,10}$/i，/[A-Z]/，/[0-9]/
~~~
