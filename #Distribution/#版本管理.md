## Git本地

#### #本地撤销

```shell
# 本地栈|还原（未提交）
git stash list|show
git stash save _comments
git stash apply|pop  stash@{_id}
git stash drop|clear stash@{_id}
```

```shell
# 撤销修改
git restore <file>
# 撤销缓存
git restore --staged <file>
# 撤销追踪
git rm --cached <file>
# 重置
git reset HEAD <file>
git checkout -- <file>
```

#### #分支

```shell
git pull
git fetch
git merge
# 创建
git branch xxxbra
git checkout xxxbra
git checkout -b xxxbra
# 删除|找回
git branch -d|-D xxxbra
git branch xxxbra xxxid
# 批量删除分支
git branch --merged
git branch --nomerged

# 合并|取消分支
git checkout xxxbra
git merge xxxbra
git reset --hard ORIG_HEAD
# 快转机制
git merge xxxbra --no-ff
git merge xxxbra --no-ff --no-commit
git merge xxxbra --squash --no-commit
git commit
# 合并冲突
git merge --abort
git add . commit(手动合并)

git checkout -b local remote
git merge bra
git branch -d/-m bra
git push origin :remote
```

```shell
git branch -r
git branch -vv
# 拉取远程并创建本地分支
git pull
git checkout -b <local> origin/<remote>
# 删除远程分支
git push origin --delete dev
```

```shell
# 合并多个Commit
git rebase -i <hash>
pick >>> squash
# 回退某版本
git revert -n <hash>
git add .
git revert --continue
```

#### Git配置

```shell

# 增删改查
git config --list

git config --unset <key> <value>
git config <key> <value>
# 常用配置
- user.name
- user.email
- alias.lol "log --oneline --decorate --graph --all"
- core.ignorecase false # 大小写转换
# 忽略配置.gitignore
/node_modules | *.txt | *.zip
# 恢复初始化
rm -rf .git
```
